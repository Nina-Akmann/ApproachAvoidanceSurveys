add_likert_labels <- function(x) {
 val_labels(x) <- c("überhaupt nicht typisch für mich" = 1,
 "2" = 2,
 "einigermaßen typisch für mich" = 3,
 "4" = 4, 
 "vollkommen typisch für mich" = 5) # change to correct labels 
 x
}
